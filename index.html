function calculateCompoundInterest(principal, rate, years) {
    let balance = principal;
    
    console.log("Year 0: $" + balance.toFixed(2));
    
    for (let year = 1; year <= years; year++) {
        balance = balance * (1 + rate);
        console.log("Year " + year + ": $" + balance.toFixed(2));

    let totalInterest = balance - principal;
    return totalInterest;
}

Arrow Functions

Arrow functions provide a shorter, cleaner syntax for writing functions. They're especially useful for quick calculations and callbacks.

Traditional vs Arrow Function

Traditional Function

function calculateTax(amount, rate) {
    return amount * rate;
}

let tax = calculateTax(1000, 0.07);
console.log(tax);  // 70

Arrow Function

const calculateTax = (amount, rate) => {
    return amount * rate;
};

let tax = calculateTax(1000, 0.07);
console.log(tax);  // 70

Arrow Function Variations

// Standard arrow function
const calculateTax = (amount, rate) => {
    return amount * rate;
};

// Implicit return (one-liner)
const calculateTax2 = (amount, rate) => amount * rate;

// No parameters
const getCompanyName = () => "ABC Corp";

// One parameter (parentheses optional)
const double = num => num * 2;

// Multiple statements (need curly braces)
const calculateNetIncome = (revenue, expenses) => {
    let gross = revenue;
    let net = gross - expenses;
    return net;
};
Accounting Examples with Arrow Functions
// Calculate profit margin
const profitMargin = (netIncome, revenue) => (netIncome / revenue) * 100;

console.log(profitMargin(15000, 100000));  // 15

// Calculate compound interest
const compound = (p, r, t) => p * Math.pow(1 + r, t);

console.log(compound(10000, 0.05, 5).toFixed(2));  // 12762.82

// Array processing with arrow functions
let revenues = [50000, 55000, 48000, 62000];

// Map: apply calculation to each element
let withTax = revenues.map(rev => rev * 1.07);
console.log(withTax);  // [53500, 58850, 51360, 66340]

// Filter: keep only values meeting criteria
let highRevenue = revenues.filter(rev => rev > 50000);
console.log(highRevenue);  // [55000, 62000]

// Reduce: calculate total
let total = revenues.reduce((sum, rev) => sum + rev, 0);
console.log(total);  // 215000
When to Use Arrow Functions:
  • Array methods (map, filter, reduce)
  • Short, simple calculations
  • Callbacks and event handlers
  • When you want cleaner syntax
When NOT to use:
  • Object methods (use regular functions for proper this binding)
  • When you need dynamic this context

Default Parameters

Default parameters allow you to set default values for function parameters, making functions more flexible and reducing the need for manual checks.

Without Default Parameters (Old Way)

function calculateInterest(principal, rate, time) {
    // Manual checks for missing parameters
    if (rate === undefined) {
        rate = 0.05;  // Default 5%
    }
    if (time === undefined) {
        time = 1;     // Default 1 year
    }
    return principal * rate * time;
}

console.log(calculateInterest(1000));           // 50
console.log(calculateInterest(1000, 0.08));     // 80
console.log(calculateInterest(1000, 0.08, 2));  // 160

With Default Parameters (ES6 Way)

function calculateInterest(principal, rate = 0.05, time = 1) {
    return principal * rate * time;
}

console.log(calculateInterest(1000));           // 50 (uses defaults)
console.log(calculateInterest(1000, 0.08));     // 80 (uses default time)
console.log(calculateInterest(1000, 0.08, 2));  // 160 (all specified)
Accounting Examples with Default Parameters
// Invoice calculator with defaults
const calculateInvoice = (
    subtotal,
    taxRate = 0.07,        // Default 7% tax
    discount = 0,          // Default no discount
    shippingFee = 10       // Default $10 shipping
) => {
    let tax = subtotal * taxRate;
    let discountAmount = subtotal * discount;
    let total = subtotal + tax - discountAmount + shippingFee;
    return total;
};

console.log(calculateInvoice(100));                    // $117.00
console.log(calculateInvoice(100, 0.08));             // $118.00
console.log(calculateInvoice(100, 0.08, 0.1));        // $108.00
console.log(calculateInvoice(100, 0.08, 0.1, 0));     // $98.00

// Depreciation calculator with defaults
const depreciation = (
    cost,
    salvageValue = 0,
    usefulLife = 5,
    method = "straight-line"
) => {
    if (method === "straight-line") {
        return (cost - salvageValue) / usefulLife;
    }
    return 0;
};

console.log(depreciation(50000));              // $10000/year
console.log(depreciation(50000, 5000));        // $9000/year
console.log(depreciation(50000, 5000, 10));    // $4500/year
Best Practices for Default Parameters:
  • Place parameters with defaults at the end of the parameter list
  • Use sensible defaults that work in most common cases
  • Document what the defaults are in comments
  • Consider using objects for functions with many optional parameters

Rest and Spread Operators

Both use the ... syntax, but they work in opposite ways:

  • Rest: Collects multiple items into an array
  • Spread: Expands an array into individual items

Rest Operator (...)

The rest operator collects multiple arguments into a single array parameter.

// Sum any number of transactions
function sumTransactions(...amounts) {
    let total = 0;
    for (let amount of amounts) {
        total += amount;
    }
    return total;
}

console.log(sumTransactions(100, 200, 300));           // 600
console.log(sumTransactions(50, 75, 125, 200));        // 450
console.log(sumTransactions(1000, 500, 250, 100, 50)); // 1900

// Arrow function version
const sum = (...nums) => nums.reduce((total, n) => total + n, 0);

console.log(sum(10, 20, 30, 40, 50));  // 150

Rest with Other Parameters

// First parameter separate, rest collected
function processPayroll(companyName, ...employeeSalaries) {
    console.log(`Processing payroll for ${companyName}`);
    let totalPayroll = employeeSalaries.reduce((sum, sal) => sum + sal, 0);
    console.log(`Total Payroll: $${totalPayroll}`);
    console.log(`Number of Employees: ${employeeSalaries.length}`);
    return totalPayroll;
}

processPayroll("ABC Corp", 50000, 60000, 55000, 70000);
// Processing payroll for ABC Corp
// Total Payroll: $235000
// Number of Employees: 4

Spread Operator (...)

The spread operator expands an array into individual elements.

// Combining arrays
let q1Revenue = [50000, 55000, 48000];
let q2Revenue = [60000, 62000, 58000];

let halfYear = [...q1Revenue, ...q2Revenue];
console.log(halfYear);  
// [50000, 55000, 48000, 60000, 62000, 58000]

// Copying arrays
let originalData = [100, 200, 300];
let copyData = [...originalData];
copyData[0] = 999;

console.log(originalData);  // [100, 200, 300] - unchanged
console.log(copyData);      // [999, 200, 300] - modified

// Using Math functions with arrays
let expenses = [5000, 3200, 4800, 6100];

let maxExpense = Math.max(...expenses);
let minExpense = Math.min(...expenses);

console.log("Highest Expense: $" + maxExpense);  // $6100
console.log("Lowest Expense: $" + minExpense);   // $3200
Practical Accounting Examples
// Calculate portfolio value from multiple accounts
const calculatePortfolio = (...accountBalances) => {
    const total = accountBalances.reduce((sum, bal) => sum + bal, 0);
    const average = total / accountBalances.length;
    const max = Math.max(...accountBalances);
    const min = Math.min(...accountBalances);
    
    return {
        total: total,
        average: average,
        highest: max,
        lowest: min,
        count: accountBalances.length
    };
};

let portfolio = calculatePortfolio(5000, 12000, 8500, 15000, 6200);

// Combine monthly data from different sources

let wholesaleSales = [15000, 16000, 14500];

let allSales = [...onlineSales, ...storeSales, ...wholesaleSales];

Destructuring Assignment

Destructuring allows you to extract values from arrays or properties from objects into distinct variables using a concise syntax.

Array Destructuring

// Traditional way
let balances = [1000, 2000, 3000];
let checking = balances[0];
let savings = balances[1];
let investment = balances[2];

// Destructuring way
let [checking2, savings2, investment2] = balances;

console.log(checking2);    // 1000
console.log(savings2);     // 2000
console.log(investment2);  // 3000

// Skip elements
let [first, , third] = balances;
console.log(first);   // 1000
console.log(third);   // 3000

// With rest operator
let numbers = [10, 20, 30, 40, 50];
let [a, b, ...remaining] = numbers;

console.log(a);         // 10
console.log(b);         // 20
console.log(remaining); // [30, 40, 50]

Object Destructuring

class BankAccount {
    constructor(accountNumber, accountHolder, initialBalance = 0) {
        this.accountNumber = accountNumber;
        this.accountHolder = accountHolder;
        this.balance = initialBalance;
        this.transactions = [];
        this.createdDate = new Date();
    }
    
    deposit(amount) {
        if (amount <= 0) {
            throw new Error("Deposit amount must be positive");
        }
        this.balance += amount;
        this.recordTransaction("deposit", amount);
        return this.balance;
    }
    
    withdraw(amount) {
        if (amount <= 0) {
            throw new Error("Withdrawal amount must be positive");
        }
        if (amount > this.balance) {
            throw new Error("Insufficient funds");
        }
        this.balance -= amount;
        this.recordTransaction("withdrawal", amount);
        return this.balance;
    }
    
    recordTransaction(type, amount) {
        this.transactions.push({
            type,
            amount,
            date: new Date(),
            balanceAfter: this.balance
        });
    }
    
    getBalance() {
        return this.balance;
    }
    
    getTransactionHistory() {
        return this.transactions;
    }
    
    printStatement() {
        console.log(`\n╔════════════════════════════════════════╗`);
        console.log(`║     Account Statement                  ║`);
        console.log(`╠════════════════════════════════════════╣`);
        console.log(`║ Account: ${this.accountNumber.padEnd(28)}║`);
        console.log(`║ Holder:  ${this.accountHolder.padEnd(28)}║`);
        console.log(`║ Balance: $${String(this.balance.toFixed(2)).padEnd(27)}║`);
        console.log(`╠════════════════════════════════════════╣`);
        console.log(`║ Transactions:                          ║`);
        
        this.transactions.forEach(trans => {
            let type = trans.type.padEnd(12);
            let amt = `$${trans.amount.toFixed(2)}`.padEnd(10);
            console.log(`║ ${type} ${amt}              ║`);
        });
        
        console.log(`╚════════════════════════════════════════╝\n`);
    }
}

// Base class
class Account {
    constructor(accountNumber, balance = 0) {
        this.accountNumber = accountNumber;
        this.balance = balance;
    }
    
    deposit(amount) {
        this.balance += amount;
        return this.balance;
    }

        return this.balance;
    }
}

// Savings Account (extends Account)
class SavingsAccount extends Account {
    constructor(accountNumber, balance, interestRate) {
        super(accountNumber, balance);  // Call parent constructor
        this.interestRate = interestRate;
    }
    
    addInterest() {
        let interest = this.balance * this.interestRate;
        this.balance += interest;
        return interest;
    }
}

// Checking Account (extends Account)
class CheckingAccount extends Account {
    constructor(accountNumber, balance, overdraftLimit) {
        super(accountNumber, balance);
        this.overdraftLimit = overdraftLimit;
    }
    
    withdraw(amount) {
        if (amount > this.balance + this.overdraftLimit) {
            throw new Error("Exceeds overdraft limit");
        }
        this.balance -= amount;
        return this.balance;
    }
}

Getters and Setters

Getters and setters allow you to define methods that are accessed like properties, providing controlled access to object data.

Basic Getters and Setters

class Temperature {
    constructor(fahrenheit) {
        this._fahrenheit = fahrenheit;
    }
    
    // Getter - accessed like a property
    get celsius() {
        return (this._fahrenheit - 32) * 5/9;
    }
    
    // Setter - set like a property
    set celsius(value) {
        this._fahrenheit = (value * 9/5) + 32;
    }
    
    get fahrenheit() {
        return this._fahrenheit;
    }
    
    set fahrenheit(value) {
        this._fahrenheit = value;
    }
}

let temp = new Temperature(32);
console.log(temp.celsius);     // 0 (getter)
console.log(temp.fahrenheit);  // 32

temp.celsius = 100;            // setter
console.log(temp.fahrenheit);  // 212

class FinancialStatement {

Module Best Practices

Organizing Your Code with Modules:

Object Shorthand

const name = "John";
const age = 30;

// Shorthand
const person = { name, age };

// Same as
const person = { name: name, age: age };

Concise Methods

const obj = {
    // Concise method
    method(param) {
        return param;
    },
    
    // Traditional
    method2: function(param) {
        return param;
    }
};

Classes

class ClassName {
    constructor(param) {
        this.property = param;
    }
    
    method() {
        return this.property;
    }
    
    get computed() {
        return this.property * 2;
    }
    
    set value(val) {
        this.property = val;
    }
}

    }
}

Import/Export

// Named exports
export const value = 42;
export function func() {}

// Default export
export default class MyClass {}

// Named imports
import { value, func } from './module.js';

// Default import
import MyClass from './module.js';

// Import all
import * as Module from './module.js';

// Rename
import { value as val } from './module.js';
